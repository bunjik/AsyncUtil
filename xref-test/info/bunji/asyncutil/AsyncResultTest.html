53  	@Test
54  	public void testClose() throws Exception {
55  		AsyncResult<String> asyncResult = spy(AsyncExecutor.execute(new StringProcess1(500)));
56  
57  		@SuppressWarnings("unchecked")
58  		Subscriber<String> subscriber = (Subscriber<String>) Whitebox.getInternalState(asyncResult, "subscriber");
59  		subscriber = PowerMockito.spy(subscriber);
60  		Whitebox.setInternalState(asyncResult, "subscriber", subscriber);
61  		when(subscriber.isUnsubscribed()).thenReturn(false);
62  
63  		asyncResult.close();
64  
65  		verify(asyncResult, times(1)).close();
66  		// called unsubscribe()
67  		verify(subscriber, times(1)).unsubscribe();
68  	}

101 		try (AsyncResult<String> result = AsyncExecutor.execute(new StringProcess1(1000))) {
102 			Iterator<String> it = result.iterator();
103 			while(it.hasNext()) {
104 				it.next();
105 				count++;
106 			}
107 		}
108 		assertThat(count, is(1000));
